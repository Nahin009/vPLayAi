🌐 Frontend (Next.js App Router)

1. API Route: app/api/generate/route.ts

ts
import { NextResponse } from 'next/server'
import fs from 'fs'
import path from 'path'
import FormData from 'form-data'
import fetch from 'node-fetch'

export async function POST() {
  const audioPath = path.join(process.cwd(), 'public', 'sample.mp3')
  const subtitlePath = path.join(process.cwd(), 'public', 'subtitles.vtt')

  const form = new FormData()
  form.append('audio', fs.createReadStream(audioPath))
  form.append('subtitle', fs.createReadStream(subtitlePath))

  const response = await fetch('http://localhost:8000/process/', {
    method: 'POST',
    body: form,
    headers: form.getHeaders(),
  })

  if (!response.ok) {
    return NextResponse.json({ error: 'Failed to process subtitle' }, { status: 500 })
  }

  const buffer = await response.arrayBuffer()
  return new Response(buffer, {
    headers: {
      'Content-Type': 'text/vtt',
      'Content-Disposition': 'attachment; filename="processed.vtt"',
    },
  })
}

🧩 Updated Frontend Code

API Route: app/api/generate/route.ts

ts
import { NextRequest, NextResponse } from 'next/server'
import fs from 'fs'
import path from 'path'
import FormData from 'form-data'
import ffmpeg from 'fluent-ffmpeg'
import { promisify } from 'util'

const generateAudio = (videoPath: string, audioPath: string): Promise<void> => {
  return new Promise((resolve, reject) => {
    ffmpeg(videoPath)
      .output(audioPath)
      .audioCodec('libmp3lame')
      .on('end', () => resolve())
      .on('error', err => reject(err))
      .run()
  })
}

export async function POST(req: NextRequest) {
  const { uuid } = await req.json()

  const baseDir = path.join(process.cwd(), 'public', 'asset', 'videodata')
  const videoPath = path.join(baseDir, 'videos', `${uuid}.mp4`)
  const subtitlePath = path.join(baseDir, 'subtitles', `${uuid}.vtt`)
  const audioPath = path.join(baseDir, 'audios', `${uuid}.mp3`)

  // Generate audio if it doesn't exist
  if (!fs.existsSync(audioPath)) {
    try {
      await generateAudio(videoPath, audioPath)
    } catch (error) {
      return NextResponse.json({ error: 'Failed to extract audio' }, { status: 500 })
    }
  }

  // Create FormData for backend
  const form = new FormData()
  form.append('audio', fs.createReadStream(audioPath))
  form.append('subtitle', fs.createReadStream(subtitlePath))

  const response = await fetch('http://localhost:8000/process/', {
    method: 'POST',
    body: form,
    headers: form.getHeaders(),
  })

  if (!response.ok) {
    return NextResponse.json({ error: 'Failed to process subtitle' }, { status: 500 })
  }

  // Save result to target path
  const highlightPath = path.join(
    process.cwd(),
    'public',
    'asset',
    'highlightsdata',
    'short_subtitle',
    `${uuid}.vtt`
  )

  const buffer = await response.arrayBuffer()
  await fs.promises.mkdir(path.dirname(highlightPath), { recursive: true })
  await fs.promises.writeFile(highlightPath, Buffer.from(buffer))

  return NextResponse.json({ message: 'Subtitle processed successfully' })
}

export default function GenerateButton() {

    setStatus('Processing...')

      headers: {

      },

      setStatus('Success! Subtitle saved.')
    } else {
      setStatus('Failed to process subtitle.')
    }
  }

  return (
    <div>
      <input
        type="text"
        placeholder="Enter UUID"
        value={uuid}
        onChange={(e) => setUuid(e.target.value)}
        className="border px-2 py-1 mr-2"
      />
      <button onClick={handleClick} className="bg-blue-500 text-white px-3 py-1 rounded">
        Generate Subtitle
      </button>
      <p>{status}</p>
    </div>
  )
}
